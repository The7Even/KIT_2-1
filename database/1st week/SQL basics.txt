create table song (
  id int primary key generated always as identity,
  title varchar(255) not null,
  singer varchar(100),
  rating int,
  lyrics text null
);

기본 테이블 생성 구문. 
2번줄에서 나오는 primary key (주 키)는 테이블 한 열의 대푯값 정도로 생각하면 된다. always as identity는 컴퓨터가 알아서 할당을 하라는 의미.
다시말해 2번줄을 해석하자면 id라는 주 키를 생성하되, SQL이 그 주 키를 알아서 할당하라는 의미라고 생각하면 된다.

3,4번 줄에서 사용한 varchar는 지정된 크기까지 저장하되, 동적으로 메모리를 할당하여 varchar(255)라고 하여도 실제로는 40자만 저장한다면 40자만 할당하여 저장할 수 있도록 한다.
그렇지만 속도를 위해서 255자의 공간을 싹다 할당하는 경우도 있을지도..

6번줄에 나오는 text라는 자료형은 무제한 길이의 varchar라고 생각해도 좋다. 

create table listsong (
  listid int,
  songid int,
  primary key(listid, songid),
  foreign key(listid) references playlist(id)
    on delete cascade
    --on delete restrict 
    on update cascade,
  foreign key(songid) references song(id)
    on delete cascade 
    on update cascade
);

카테고리가 여러가지 나뉘어졌을 경우 따로 생각해야 될 새로운 테이블
"카테고리 여러개여도 그냥 테이블에 열 하나 더 만들어서 거기다가 카테고리까지 넣으면 테이블 1개면 되지 않을까?"
싶지만 만약 A라는 곡이 두개의 카테고리에 중복으로 들어가있다면?
하나의 테이블 안에 곡 A가 두번 들어가야 하는 상황이 벌어진다.
검색이나 다른 기능을 구현할때 꽤 귀찮을 것이다. 특히 실무에서는 이렇게 하면 못배웠다고 생각한단다.

아무튼 위 listsong 테이블에 대해 알아보자면
우선 primary key를 생성할 때 두개를 섞어버린 이유는 카테고리 안에 노래가 1곡씩 있는것도 아니니 중복이 되어 고유성이 사라지고, 위에서 말했듯 노래 하나하나가 무조건 카테고리 하나에만 들어가있다는 보장도 없으니
A 카테고리의 B 곡 이라는 고유성을 가지게끔 두 id를 합쳐서 primary key를 만들었다.

22, 26번 줄의 foreign key는 외부의 키를 참조하게 해주는 명령어이다.
22번 줄을 기준으로 설명해보자면 listid라는 요소는 playlist라는 table의 id키를 참조하는 키임을 알려주는 명령어이다.
그리고 23, 25, 27, 28번 줄에 보이는 on delete, on update 구문은 삭제되거나 갱신되었을때, 라는 의미이다.
그래서 on delete cascade는 원래 테이블에서 요소가 삭제되면 같이 삭제해버리라는 의미이고, on update cascade는 요소의 id가 바뀐다던가, 갱신되었을경우, 같이 갱신하라는 의미이며
간단히 말해 그냥 원래 table의 요소가 변동될경우 그 변동을 따라가라는 의미라고 해석하면 된다.

추가로 cascade 말고 restrict도 있는데, 이는 on delete 뒤에만 붙는듯 하다.
참조되고 있는경우 (foriegn key등으로) 삭제될 수 없게 하는 명령어이다.

select name, title, singer from playlist 
  left join listsong on playlist.id = listsong.listid
  left join song on listsong.songid = song.id;

해당 명령어를 이용하면 playlist, song에 맞게 id가 아닌 문자로 table이 표기된다.
select name, title, singer from playlist 라는 명령어 줄이 있는데,

name은 playlist, title, singer는 song table의 요소이므로
